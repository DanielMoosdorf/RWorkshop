5.1 Korrelationsmatrix

Wtd.cor()

# Datensatz erstellen

ds <- allbus2018 %>% 
  select(im19:im21, id01, wghtpew)

#Berechnung der Korrelationsmatrix

corMatrix <- wtd.cor(ds[1:4], weight = ds$wghtpew)
